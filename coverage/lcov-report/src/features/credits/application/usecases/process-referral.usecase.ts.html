import { CreditTransaction } from '../../domain/entities/credit-transaction.entity';
import { ICreditRepository } from '../../domain/interfaces/credit-repository.interface';
import { 
  InvalidReferralCodeError,
  SelfReferralError,
  ReferralAlreadyUsedError 
} from '../../domain/errors/credit-errors';
 
export class ProcessReferralUseCase {
  private readonly REFERRER_BONUS = 10; // Credits für den Einladenden
  private readonly REFERRED_BONUS = 5;  // Credits für den Eingeladenen
  
  constructor(
    private creditRepository: ICreditRepository
  ) {}
 
  async execute(
    referrerId: string, 
    referredUserId: string
  ): Promise<{ 
    referrerTransaction: CreditTransaction; 
    referredTransaction: CreditTransaction 
  }> {
    // Validierung der Eingabedaten
    if (!referrerId || !referredUserId) {
      throw new Error('ReferrerID und ReferredUserID sind erforderlich');
    }
 
    // Selbst-Empfehlung verhindern
    if (referrerId === referredUserId) {
      throw new SelfReferralError(referredUserId);
    }
 
    // Prüfen ob der eingeladene User bereits einen Referral-Code verwendet hat
    const hasUsedReferral = await this.creditRepository.hasUsedReferralCode(referredUserId);
    if (hasUsedReferral) {
      throw new ReferralAlreadyUsedError(referredUserId, 'ALREADY_USED');
    }
 
    // Prüfen ob der Referrer existiert (durch Referral-Code-Lookup)
    const referrerExists = await this.creditRepository.getUserByReferralCode(referrerId);
    if (!referrerExists) {
      throw new InvalidReferralCodeError(referrerId);
    }
 
    const currentTime = new Date();
 
    // Credits für den Einladenden hinzufügen
    const referrerTransaction = await this.creditRepository.createTransaction({
      userId: referrerId,
      amount: this.REFERRER_BONUS,
      transactionType: 'referral',
      status: 'completed',
      description: `Empfehlungsbonus - Neuer User eingeladen`,
      metadata: {
        referredUserId,
        reason: 'referrer_bonus'
      }
    });
 
    // Credits für den Eingeladenen hinzufügen
    const referredTransaction = await this.creditRepository.createTransaction({
      userId: referredUserId,
      amount: this.REFERRED_BONUS,
      transactionType: 'referral',
      status: 'completed',
      description: `Willkommensbonus - Empfehlung erhalten`,
      metadata: {
        reason: 'referred_bonus'
      }
    });
 
    // Referral als verwendet markieren
    await this.creditRepository.processReferral(
      referrerId, 
      referrerId, // Als Referral-Code verwenden wir die referrerId
      this.REFERRER_BONUS
    );
 
    return {
      referrerTransaction,
      referredTransaction
    };
  }

   * Berechnet die Referral-Boni basierend auf Benutzerebene oder anderen Faktoren

    return {
      referrerBonus: this.REFERRER_BONUS,
      referredBonus: this.REFERRED_BONUS
    };
  }
